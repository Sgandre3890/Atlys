cmake_minimum_required(VERSION 3.31)

project(OpenGl)

set(CMAKE_BUILD_TYPE Release)

set(CMAKE_CXX_FLAGS "${CMAKE_CXX_FLAGS} -m64")
set(CMAKE_C_FLAGS "${CMAKE_C_FLAGS} -m64")

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)

set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_SOURCE_DIR})

include_directories(${CMAKE_SOURCE_DIR}/includes)
include_directories(${CMAKE_SOURCE_DIR}/includes/learnopengl)
include_directories(${CMAKE_SOURCE_DIR}/resources/skybox)

find_package(assimp REQUIRED)

file(GLOB SOURCES "src/*.cpp")

add_executable(out ${SOURCES})

target_compile_definitions(out PRIVATE PROJECT_ROOT="${CMAKE_SOURCE_DIR}")

target_link_libraries(out PRIVATE assimp::assimp)

if (CMAKE_SYSTEM_NAME STREQUAL "Windows")
    target_link_libraries(out PRIVATE "${CMAKE_SOURCE_DIR}/statLib/libglfw3WIN.a")
    target_link_libraries(out PRIVATE "${CMAKE_SOURCE_DIR}/statLib/libgladWIN.a")
else ()
    target_link_libraries(out PRIVATE
        "-framework CoreFoundation"
        "-framework IOKit"
        "-framework Cocoa"
        "${CMAKE_SOURCE_DIR}/statLib/libglfw3MAC.a"
        "${CMAKE_SOURCE_DIR}/statLib/libgladMAC.a"
    )
endif()
